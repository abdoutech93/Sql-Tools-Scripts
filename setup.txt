#!/bin/bash

# --- 1. INITIAL INSTALLATIONS ---
echo "Updating packages and installing software..."
sudo apt update
sudo apt install mariadb-server chrony openssh-server python-is-python3 -y

# --- 2. FIREWALL CONFIGURATION ---
echo "Configuring firewall rules..."
sudo ufw allow 3306/tcp
sudo ufw allow 6000:6010/tcp
sudo ufw allow 12000/tcp
sudo ufw allow ssh
sudo ufw --force enable

# --- 3. MARIADB REMOTE ACCESS ---
echo "Opening MariaDB for remote connections..."
# Changes 127.0.0.1 to 0.0.0.0 in the config file
sudo sed -i 's/bind-address            = 127.0.0.1/bind-address            = 0.0.0.0/' /etc/mysql/mariadb.conf.d/50-server.cnf
sudo systemctl restart mariadb

# Creates the remote user (Change 'your_password' to your actual password)
sudo mysql -e "GRANT ALL PRIVILEGES ON *.* TO 'user'@'%' IDENTIFIED BY 'abdoutech93'; FLUSH PRIVILEGES;"

# --- 4. STATIC IP CONFIGURATION (NETPLAN) ---
echo "Setting static IP 192.168.88.1 on enp1s0..."
sudo bash -c 'cat << EOF > /etc/netplan/50-cloud-init.yaml
network:
  ethernets:
    enp1s0:
      dhcp4: false
      addresses:
        - 192.168.88.1/24
      nameservers:
        addresses: [8.8.8.8, 8.8.4.4]
    enp2s0:
      dhcp4: true
  version: 2
EOF'

sudo netplan apply

echo "------------------------------------------------"
echo "✅ SETUP COMPLETE!"
echo "IP: 192.168.88.1"
echo "Python: 'python' command is now active"
echo "Database: Remote access granted for 'user'"
echo "------------------------------------------------"
